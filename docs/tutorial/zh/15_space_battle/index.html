    15 — 飞机大战¶

    示例文件： examples/15_space_battle.py

    之前每个例子都把 tensor 当「可视化工具」用。但 tensor 说到底只是一张二维数字 格子——谁规定不能拿它当帧缓冲来画游戏呢？

    这一章我们用一个 160×200 的 GPU tensor 跑一个完整的太空射击游戏。所有像素 直接写到 tensor 里，每帧上传显示——键盘操控、碰撞检测、爆炸特效、录制回放， 一应俱全。

    这次玩什么¶

    一个复古像素射击游戏，160×200 分辨率：

    区域 内容
    左侧 控制 — 分数、生命、难度滑块、录制按钮
    右侧 游戏 — 像素风大战，最近邻放大显示

    操作：W/A/S/D 移动，空格 开火。

    新朋友¶

    新东西 干什么的 为什么重要
    ui.is_key_down(key) 检测键盘按键是否按下 不用轮询事件，直接读实时状态
    ImGui 键码 KEY_W = 568 之类的常量 imgui.h 枚举，512 起始
    filter="nearest" 画布用最近邻插值放大 像素清晰、不糊
    canvas.save(path) 把当前帧存成图片 一键截图
    Tensor 当帧缓冲 直接往切片里写 RGB 值 画精灵、粒子、什么都行

    键盘输入¶

    Vultorch 通过 ui 模块暴露 ImGui 的键盘状态：

    from vultorch import ui

KEY_W = 568   # imgui.h: ImGuiKey_W
KEY_A = 546
KEY_S = 564
KEY_D = 549
KEY_SPACE = 525

if ui.is_key_down(KEY_A):
    player_x -= speed
if ui.is_key_down(KEY_D):
    player_x += speed

    键码来自 imgui.h 的 ImGuiKey 枚举，命名键从 512 开始。常用的：

    键 码 键 码
    W 568 空格 525
    A 546 Esc 527
    S 564 回车 528
    D 549 Tab 512

    像素风精灵¶

    用 ASCII 画定义精灵，再解析成坐标列表：

    PLAYER_SPRITE = [
    "    X    ",
    "   XXX   ",
    "  XXXXX  ",
    " XX X XX ",
    " XXXXXXX ",
    "XXX   XXX",
    "X       X",
]

def parse_sprite(lines):
    pts = []
    for r, line in enumerate(lines):
        for c, ch in enumerate(line):
            if ch == "X":
                pts.append((r, c))
    return pts

    画的时候把坐标戳进 tensor：

    for dr, dc in PLAYER_PTS:
    r, c = player_y + dr, player_x + dc
    if 0 <= r < H and 0 <= c < W:
        fb[r, c, 0] = 0.2   # R
        fb[r, c, 1] = 0.8   # G
        fb[r, c, 2] = 1.0   # B

    最近邻渲染¶

    160×200 的画面如果用双线性插值放大会糊成一团。加 filter="nearest" 就 像素锐利了：

    canvas = panel.canvas("game", filter="nearest")

    录制回放¶

    上一章学的东西，加一行：

    ctrl.record_button(canvas, "gameplay.gif", fps=30)

    打了一局好的？停止录制，就得到一个像素完美的 GIF。

    游戏架构¶

    Game 类管理全部状态：

    • 玩家： 位置、移动速度
    • 子弹： [x, y] 列表，每帧向上 4 像素
    • 敌人： [x, y, hp] 列表，从顶部刷新向下移动
    • 爆炸： [x, y, timer] 列表，扩展后消散
    • 星星： 背景视差滚动

    每帧：game.update(move_x, move_y, shoot) → game.render(display[:, :, :3])。 Tensor 就是帧缓冲。

    完整代码¶

    import random
import torch
import vultorch
from vultorch import ui

if not torch.cuda.is_available():
    raise RuntimeError("This example needs CUDA")

device = "cuda"

GW, GH = 160, 200
PLAYER_W, PLAYER_H = 9, 7
BULLET_W, BULLET_H = 1, 4
ENEMY_W, ENEMY_H = 7, 5

KEY_W = 568
KEY_A = 546
KEY_S = 564
KEY_D = 549
KEY_SPACE = 525

C_BG       = (0.05, 0.05, 0.12)
C_PLAYER   = (0.2, 0.8, 1.0)
C_BULLET   = (1.0, 1.0, 0.3)
C_ENEMY    = (1.0, 0.3, 0.2)
C_ENEMY2   = (1.0, 0.6, 0.1)
C_STAR     = (0.6, 0.6, 0.7)
C_EXPLODE  = (1.0, 0.9, 0.4)

PLAYER_SPRITE = [
    "    X    ",
    "   XXX   ",
    "  XXXXX  ",
    " XX X XX ",
    " XXXXXXX ",
    "XXX   XXX",
    "X       X",
]

ENEMY_SPRITE = [
    " XXXXX ",
    "XXXXXXX",
    "XX X XX",
    "XXXXXXX",
    " X   X ",
]


def parse_sprite(lines):
    pts = []
    for r, line in enumerate(lines):
        for c, ch in enumerate(line):
            if ch == "X":
                pts.append((r, c))
    return pts


PLAYER_PTS = parse_sprite(PLAYER_SPRITE)
ENEMY_PTS = parse_sprite(ENEMY_SPRITE)


class Game:
    def __init__(self):
        self.px, self.py = GW // 2, GH - 20
        self.speed = 2
        self.bullets, self.enemies, self.explosions = [], [], []
        self.stars = [(random.randint(0, GW-1), random.randint(0, GH-1),
                       random.uniform(0.3, 1.0)) for _ in range(60)]
        self.score, self.lives = 0, 3
        self.game_over = False
        self.spawn_timer, self.spawn_rate = 0, 30
        self.bullet_cooldown, self.frame = 0, 0
        self.difficulty = 1.0

    def reset(self):
        self.__init__()

    def update(self, move_x, move_y, shoot):
        if self.game_over:
            return
        self.frame += 1
        self.px = max(PLAYER_W//2, min(GW-1-PLAYER_W//2, self.px + move_x*self.speed))
        self.py = max(PLAYER_H, min(GH-5, self.py + move_y*self.speed))
        if shoot and self.bullet_cooldown <= 0:
            self.bullets.append([self.px, self.py - PLAYER_H//2 - 1])
            self.bullet_cooldown = 6
        self.bullet_cooldown = max(0, self.bullet_cooldown - 1)

        self.bullets = [b for b in self.bullets if (b.__setitem__(1, b[1]-4) or b[1] > 0)]

        self.spawn_timer -= 1
        if self.spawn_timer <= 0:
            ex = random.randint(ENEMY_W//2, GW-1-ENEMY_W//2)
            self.enemies.append([ex, -ENEMY_H, 1])
            self.spawn_timer = max(8, int(self.spawn_rate / self.difficulty))

        # ... (碰撞检测、敌人移动、爆炸、星星)

    def render(self, fb):
        fb[:,:,0] = C_BG[0]; fb[:,:,1] = C_BG[1]; fb[:,:,2] = C_BG[2]
        # 画星星、敌人、子弹、爆炸、玩家 ...
        if self.game_over:
            fb *= 0.3


game = Game()
view = vultorch.View("15 - Space Battle", 900, 750)
ctrl = view.panel("Controls", side="left", width=0.30)
game_panel = view.panel("Game")

display = vultorch.create_tensor(GH, GW, channels=4, device=device,
                                 name="game", window=view.window)
display[:, :, 3] = 1.0
canvas = game_panel.canvas("game", filter="nearest")
canvas.bind(display)


@view.on_frame
def update():
    move_x = (1 if ui.is_key_down(KEY_D) else 0) - (1 if ui.is_key_down(KEY_A) else 0)
    move_y = (1 if ui.is_key_down(KEY_S) else 0) - (1 if ui.is_key_down(KEY_W) else 0)
    shoot = ui.is_key_down(KEY_SPACE)
    game.update(move_x, move_y, shoot)
    with torch.no_grad():
        game.render(display[:, :, :3])


@ctrl.on_frame
def draw_ctrl():
    global game
    ctrl.text(f"FPS: {view.fps:.0f}")
    ctrl.separator()
    ctrl.text(f"Score: {game.score}")
    hearts = "♥ " * game.lives + "♡ " * (3 - game.lives)
    ctrl.text(f"Lives: {hearts}")
    if game.game_over:
        ctrl.text_colored(1, 0.3, 0.3, 1, "GAME OVER!")
        if ctrl.button("Restart", width=150):
            game = Game()
    ctrl.separator()
    game.difficulty = ctrl.slider("Difficulty", 0.5, 3.0, default=1.0)
    game.speed = ctrl.slider_int("Player Speed", 1, 5, default=2)
    ctrl.separator()
    ctrl.record_button(canvas, "gameplay.gif", fps=30)
    if canvas.is_recording:
        ctrl.text_colored(1, 0.3, 0.3, 1, "● REC")
    ctrl.separator()
    if ctrl.button("Screenshot", width=150):
        canvas.save("screenshot.png")


view.run()

    完整代码在 examples/15_space_battle.py，包含完整的 Game 类：碰撞检测、 爆炸特效、星空滚动等。
