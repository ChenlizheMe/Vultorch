from vultorch import ui

texture = torch.rand(512, 512, 4, device="cuda")

win = vultorch.Window("Viewer", 800, 600)

    ui.begin("Output")
    vultorch.show(texture)  # zero-copy GPU â†’ screen
    ui.end()
    win.end_frame()
